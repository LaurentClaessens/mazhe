\begin{exercice}\label{exoDerrivePartielle-0003}

	(INGE1121, 9.12) Calculer $\frac{ \partial w }{ \partial x }$ et $\frac{ \partial w }{ \partial y }$ dans les cas suivants:
	\begin{enumerate}

		\item
			$w=u\cos(v)$, $u=x^2+y^2$, $v=xy$.
		\item
			$w=2uv+3v^2$, $u=x\sin(y)$, $v=y\sin(x)$.

	\end{enumerate}

\end{exercice}
