\begin{exercice}\label{exoEqsDiff0015}

Déterminer une solution particulière de l'équation
$y''+y=q$ pour

\begin{enumerate}

\item $ q = x^3$
\item $ q = \sinh x$
\item $ q = 1/\sin x$
\end{enumerate}

\end{exercice}
