\begin{exercice}\label{exoLineraire0008}

Soit
\begin{equation}
	A=\begin{pmatrix}
	2	&	1	\\ 
	2	&	1	
\end{pmatrix}.
\end{equation}
Trouver toutes les matrices $X$ telles que $AX=A$.

\end{exercice}
