\begin{exercice}\label{exoDS2010-1-0004}

Soit $(u_n)_{n\in\mathbb{N}}$ la suite définie par 
\begin{equation}
 \left\{ \begin{array}{ll}
    u_{n+1}=u_n+2^{n},\\
    u_0=0.
  \end{array}\right.
\end{equation}
\begin{enumerate}
\item Démontrer que $\displaystyle u_n=2^n-1$.
\item Calculer, si elle existe, la limite de la suite $(u_n)_{n}$.
\end{enumerate}

\end{exercice}
