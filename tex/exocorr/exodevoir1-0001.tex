$ A_2 = \{ ( t , \sin(1/t) ) \in \eR^2 \; | \; t \in ]0, 1/2] \; \} ; $

$A_3 = \eN \times \eQ = \{ (x, y ) \in \eR^2 \; | \; x \in \eN , y \in \eQ \}. $
	\end{enumerate}
