\begin{exercice}\label{exoOutilsMath-0125}

    On considère la fonction
    \begin{equation}
        f\colon (x,y)\in\eR^2\to (x-iy)^2
    \end{equation}
    où \( i=\sqrt{-1}\).
    \begin{enumerate}
        \item
            Vérifier que les parties réelles et imaginaires de \( f(x,y)\) sont données par
            \begin{equation}
                \begin{aligned}[]
                    Re\, f(x,y)&=x^2-y^2\\
                    Im\, f(x,y)&=-2xy.
                \end{aligned}
            \end{equation}
        \item
            Montrer que le champ
            \begin{equation}
                F(x,y)=\begin{pmatrix}
                    Re\,f(x,y)    \\ 
                    Im\, f(x,y)    
                \end{pmatrix}
            \end{equation}
            est à divergence nul et rotationnel nul.
    \end{enumerate}

\end{exercice}
