\begin{exercice}\label{exoFoncDeuxVar0007}

Écrire les équations de la droite tangente à la courbe
\begin{subequations}
\begin{numcases}{}
3x^2y+y^2z=2\\
2xz-x^2y=3
\end{numcases}
\end{subequations}
au point $(1,-1,1)$.

\end{exercice}
