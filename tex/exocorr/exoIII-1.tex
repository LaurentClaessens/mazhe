\begin{exercice}\label{exo1}

Calculer
\begin{enumerate}

\item
$f'(2)$ si $f(x)=\ln(x^2+1)$
\item
$g'(e)$ si $g(x)=x^x$
\item
$h'(\frac{ \pi }{ 6 })$ si $h(x)=x e^{\sin(x)}$
\item
$\ell'(\frac{ \pi }{ 3 })$ si $\ell(x)=\frac{ x }{ 5+\cos(x) }$.

\end{enumerate}

\end{exercice}
