\begin{exercice}\label{exoOutilsMath-0106}

    Soit $D=\{ (x,y)\in \eR^2\tq x^2+y^2-2x\leq 0 \}$.
    \begin{enumerate}
        \item
            Montrer que $D$ est un disque. Préciser le centre et le rayon.
        \item
            Calculer $\iint_D\sqrt{x^2+y^2}dxdy$.
    \end{enumerate}

\end{exercice}
