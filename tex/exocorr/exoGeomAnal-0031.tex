\begin{exercice}\label{exoGeomAnal-0031}

Trouver, si elle existe, une fonction $g: \eR^3\to\eR$ telle que 
\begin{equation}
  \begin{aligned}
    &\partial_xg(x,y,z)= yz+y\cos(xy), \\
    &\partial_yg(x,y,z)=  xz+x\cos(xy),\\
    &\partial_zg(x,y,z)=xy,\\
    & g(0,1,2) =5.
  \end{aligned}
\end{equation}

\end{exercice}
