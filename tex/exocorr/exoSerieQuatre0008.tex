\begin{enumerate}

	\item
		$\cos(x)$ pour $x\simeq\frac{\pi}{2}$
	\item
		$\sqrt{x+1}-\sqrt{x}$ pour $x$ grand
	\item
		$\sqrt[6]{1+x}-1$ pour $x\simeq 0$.

\end{enumerate}
