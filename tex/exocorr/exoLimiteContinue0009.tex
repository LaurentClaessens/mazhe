\begin{exercice}[\boringexo]\label{exoLimiteContinue0009}

Étudier la continuité  des  fonctions suivantes~:
\begin{enumerate}
         
\item $f(x,y) = \left\{
         \begin{array}{cc}
         \frac{x^4y}{x^4+y^6} & \mbox{ si }(x,y) \neq (0,0) \\
         0 & \mbox{ sinon. }
         \end{array}
         \right .$

\item $f(x,y) = \left\{
         \begin{array}{cc}
         \frac{xy^4}{x^4+y^6} & \mbox{ si }(x,y) \neq (0,0) \\
         0 & \mbox{ sinon. }
         \end{array}
         \right .$
         
\item $f(x,y) = \left\{
         \begin{array}{cc}
         y^2\sin \frac{x}{y} & \mbox{ si }y \neq 0 \\
         0 & \mbox{ sinon. }
         \end{array}
         \right .$
         
\item $f(x,y) = \left\{
         \begin{array}{cc}
         xe^{\arctan \frac{y}{x}} & \mbox{ si }x \neq 0 \\
         0 & \mbox{ sinon. }
         \end{array}
         \right .$
         
\end{enumerate}


\corrref{LimiteContinue0009}
\end{exercice}
