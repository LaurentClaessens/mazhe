Soit $ A = \{( x, y) \in \eR^2 \; | \; x>0, y > 0 \}. $ 
\begin{enumerate}
	\item
		Montrer que $A$ est ouvert dans $\eR^2$.
	\item
		Trouver son adhérence $\overline{A}$ 
	\item
		Donner sa frontière $\partial A$. 

\end{enumerate}
