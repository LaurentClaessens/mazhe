\begin{exercice}\label{exoreserve0004}

Résoudre les équations suivantes :
\begin{enumerate}
\item $ty'+y=ty^2$,
\item $y'=\frac{ y }{ t }\big( y\ln| t |-1 \big)$,
\item\label{ItemCII107} $y-\cos(t)y'=\cos(t)\big(1-\sin(t)\big)y^2$.
\end{enumerate}

\end{exercice}
