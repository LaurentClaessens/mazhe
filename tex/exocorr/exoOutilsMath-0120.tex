\begin{exercice}\label{exoOutilsMath-0120}

    Soit
    \begin{equation}
        \Omega=\{ (x,y);x\geq 0,y\geq 0,1\leq x^2+y^2\leq 4 \}.
    \end{equation}
    Dessiner \( \Omega\) puis calculer
    \begin{equation}
        I=\iint_{\Omega}\frac{ xy^2+1 }{ \sqrt{x^2+y^2} }dxdy
    \end{equation}
    en utilisant les coordonnées polaires.

\end{exercice}
