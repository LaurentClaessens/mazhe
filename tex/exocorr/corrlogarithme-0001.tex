\begin{corrige}{logarithme-0001}

	\begin{enumerate}
		\item
			$3$ parce que $2^3=8$;
		\item
			$4$ parce que $10^{4}=10000$;
		\item
			$-2$ en utilisant le fait que $1/16=16^{-1}$ et que $\log_4(x^{-1})=-\log_4(x)$, puis le fait que $16=4^2$;
		\item
			$\frac{ 3 }{ 2 }$ parce que $27=3^3=(\sqrt{9})^3=9^{3/2}$.
	\end{enumerate}

\end{corrige}
