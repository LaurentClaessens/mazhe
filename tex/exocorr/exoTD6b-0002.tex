\begin{exercice}\label{exoTD6b-0002}

Déterminez l'ensemble des fonctions qui satisfont les équations différentielles suivantes.

\begin{enumerate}
	\item $ y' = (t^3 + t) e^{-y} $
	\item $ y' = 1 + y^2 $
	\item $ y' = \frac{\cos(t)}{1 + e^y} $
	\item $ y' = y^2 $
	\item $ y' = y^{\frac{1}{3}} $
	
\end{enumerate}

\end{exercice}
