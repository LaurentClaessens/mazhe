\begin{exercice}\label{exoINGE1121La0007}

	En n'utilisant que les propriétés des déterminants, démontrer l'égalité
	\begin{equation}
		\det\begin{pmatrix}
			1	&	a	&	a^2	\\
			1	&	b	&	b^2	\\
			1	&	c	&	c^2
		\end{pmatrix}.
	\end{equation}
	Ce déterminant est le \wikipedia{fr}{Matrice_de_Vandermonde}{déterminant de Vandermonde}.

\end{exercice}
