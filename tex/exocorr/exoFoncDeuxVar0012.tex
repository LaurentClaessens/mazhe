\begin{exercice}\label{exoFoncDeuxVar0012}

	(INGE1121 8.3) Étudier la continuité des fonctions suivantes:
	\begin{enumerate}

		\item
			\begin{equation}
				f(x,y)=\begin{cases}
					\frac{ 2xy }{ x^2+y^2 }	&	\text{si }(x,y)\neq(0,0)\\
					0	&	 \text{si }(x,y)=(0,0)
				\end{cases}
			\end{equation}
		\item
			\begin{equation}
				f(x,y)=\begin{cases}
					\frac{ y^2-x^2 }{ y^4+x^2 }	&	\text{si }(x,y)\neq(0,0)\\
					0	&	 \text{si }(x,y)=(0,0)
				\end{cases}
			\end{equation}
			
		\item
			\begin{equation}
				f(x,y)=\begin{cases}
					y\cos(\frac{1}{ y })	&	\text{si }y\neq 0\\
					0	&	 \text{si }y=0
				\end{cases}
			\end{equation}
	\end{enumerate}

\corrref{FoncDeuxVar0012}
\end{exercice}
