\begin{exercice}[\boringexo]\label{exoTD1_4}

Soit $f$ la fonction $x\mapsto x^3$. Dessinez les graphes des fonctions 
\begin{enumerate}
\item $f(x)-1$ ;
  \item $f(x+3)$ ;
    \item  $f(x-1)$ ;
      \item  $-f(x)$ ;
        \item  $f(-x)$ ;
          \item $|f(x)|$ ;
            \item $\min\{f(x), 0\}$ ;
              \item $\max\{1, f(x)\}$.
\end{enumerate}
