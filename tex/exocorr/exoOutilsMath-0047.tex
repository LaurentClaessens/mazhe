\begin{exercice}\label{exoOutilsMath-0047}

    Soit $v=(1,2,3)$.
    \begin{enumerate}
        \item
            Donner un vecteur non nul perpendiculaire à $v$;
        \item
            Donner un vecteur de norme $1$ perpendiculaire à $v$.
    \end{enumerate}
