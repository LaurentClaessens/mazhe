\begin{exercice}\label{exo116}

Étudier la convergence sur $[0,1]$ de la suite $f_n(x)=nx e^{-nx^2}$. Calculer
\begin{equation}
	\begin{aligned}[]
		\lim_{n\to\infty}\int_0^1f_n(x)dx	&&	\text{et}	&&	\int_0^1\lim_{n\to\infty}f_n(x)dx.
	\end{aligned}
\end{equation}

\end{exercice}
