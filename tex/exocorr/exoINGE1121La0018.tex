\begin{exercice}\label{exoINGE1121La0018}

	(INGE1121, 6.5) Sachant que $(-1,0,1,0)$ est un vecteur propre de la matrice
	\begin{equation}
		A=\begin{pmatrix}
			 2	&	1	&	-1	&	1	\\
			 1	&	0	&	1	&	1	\\
			 -1	&	1	&	2	&	1	\\ 
			 1	&	1	&	1	&	0	 
		 \end{pmatrix}
	\end{equation}
	\begin{enumerate}

		\item
			Diagonaliser $A$ au moyen d'une matrice orthogonale
		\item
			Écrire la forme quadratique $X^tAX$ sous forme d'une somme pondérée de carrés.
	\end{enumerate}

\end{exercice}
