\begin{exercice}\label{exoFoncDeuxVar0017}

	(INGE1121, 9.13) Calculer $\frac{ \partial w }{ \partial r }$ et $\frac{ \partial w }{ \partial s }$ dans les cas suivants :
	\begin{enumerate}

		\item
			$w=u^2+uv$, $u=r^2$, $v=2r+s$,
		\item
			$w= e^{xy}$, $y=r+s$, $x=rs$.

	\end{enumerate}
