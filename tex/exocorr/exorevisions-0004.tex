\begin{exercice}[\boringexo]\label{exorevisions-0004}

	Calculer les limites
	\begin{enumerate}
		\item
			$\displaystyle\lim_{x\to 2} \frac{ x^2+x-6 }{ 4(x^2-4) }$
		\item
			$\lim_{x\to \infty} \ln(x+1)$
		\item
			$\displaystyle\lim_{x\to 0} \frac{ \sin(x) }{ x }$
		\item
			$\lim_{x\to \infty} x^3-2x^2$.
	\end{enumerate}
