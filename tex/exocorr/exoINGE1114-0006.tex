\begin{exercice}\label{exoINGE1114-0006}

	Déterminer l'ensemble des $x$ qui satisfont
	\begin{enumerate}

		\item
			$| x |\leq 2$
		\item
			$\left| 5-\frac{1}{ x } \right| <1$
		\item
			$\frac{1}{ | x+2 | }>\frac{1}{ | x-1 | }$
		\item
			$| x^2-2 |\leq 1$
		\item
			$| 3-2x |<1$.
	\end{enumerate}

\end{exercice}
