\begin{exercice}\label{exoTP50002}

	Trouver toutes les solutions du système
	\begin{equation}	
		\left\{
		\begin{array}{ll}
			x+2y+z-t=3\\
			2x-y+t=2\\
			5y+2z-3t=4
		\end{array}
		\right.
	\end{equation}
	en utilisant la méthode de Gauss.

\end{exercice}
