\begin{exercice}\label{exoOutilsMath-0105}

    Calculer l'intégrale double
    \begin{equation}
        \iint_D\frac{ 1 }{ 1+x^2+y^2 }dxdy
    \end{equation}
    où $D=\{ (x,y)\in\eR^2\tq 0\leq x\leq 1,0\leq y\leq 1,x^2+y^2\leq 1 \}$.

\end{exercice}
