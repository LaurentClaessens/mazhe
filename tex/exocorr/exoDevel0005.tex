\begin{exercice}\label{exoDevel0005}

Déterminez les séries de Mc Laurin (séries de Taylor autour de 0) des fonctions suivantes
\begin{multicols}{2}
\begin{enumerate}
\item $ x \mapsto \sin(x) $
\item $ y \mapsto \cos(y) $
\item $ t \mapsto e^t $
\item $ x \mapsto \ln(1+x) $
\item $ t \mapsto  \sqrt{1 + t} $
\item $ y \mapsto \frac{1}{1 +y} $
\end{enumerate}
\end{multicols}

\corrref{Devel0005}
\end{exercice}
