\begin{exercice}\label{exoEquaDiff0003}

Résoudre les équations suivantes.
\begin{enumerate}

\item
$y'+2xy=4x$
\item
$(x-2)y'=y+2(x-2)^2$
\item
$y'+y\cotg(x)=5 e^{\cos(x)}$, avec la condition initiale $x_0=\frac{ \pi }{2}$, $y_0=-4$.

\end{enumerate}

\corrref{EquaDiff0003}
\end{exercice}
