\begin{exercice}\label{exoLineraire0001}

Exercice 5, page 85. Résoudre les systèmes
\begin{multicols}{2}
\begin{enumerate}
\item

$
\left\{
\begin{array}{ll}
x+2y=4\\
2x+y=5
\end{array}
\right.
$
\item
$
\left\{
\begin{array}{ll}
x_1+2x_2+3x_3=0\\
2x_1+x_2+3x_3=0\\
3x_1+2x_2+x_3=0
\end{array}
\right.
$
\item
$
\left\{
\begin{array}{ll}
x+y+z-3t=2\\
x-2y+2z+15t=-3\\
x+y-z-9t=0
\end{array}
\right.
$

\item
$
\left\{
\begin{array}{ll}
x+4y+z=12\\
x+y-z=0\\
2x+y=4\\
x+z=4
\end{array}
\right.
$
\item
$\left\{
\begin{array}{ll}
2x+y-3z-t=1\\
x-4y+3z+4t=-4
\end{array}
\right.$
\end{enumerate}	
\end{multicols}

\corrref{Lineraire0001}
\end{exercice}
