\begin{exercice}\label{exoreserve0009}

    Soit le champ de vecteurs 
    \begin{equation}
        F(x,y,z)=\begin{pmatrix}
            x^2y    \\ 
            z    \\ 
            0    
        \end{pmatrix}.
    \end{equation}
    \begin{enumerate}
        \item
            Calculer \( \nabla\times F\).
        \item
            Est-ce que \( F\) dérive d'un potentiel ? Si oui, en donner un.

    \end{enumerate}
