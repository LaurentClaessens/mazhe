\begin{exercice}\label{exoGeomAnal-0022}

Trouver, si elle existe, une fonction $g: \eR^3\to\eR$ telle que 
\begin{equation}
  \begin{aligned}
    &\partial_xg(x,y,z)= \frac{3\sin^2(z)}{x}+2y(xy-z), \\
    &\partial_yg(x,y,z)=  2x(xy-z),\\
    &\partial_zg(x,y,z)=3\sin(2z)\ln(x)-2(xy-z),\\
    & g(0,1,\pi) =5.
  \end{aligned}
\end{equation}

\end{exercice}
