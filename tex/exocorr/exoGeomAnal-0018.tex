\begin{exercice}\label{exoGeomAnal-0018}

  \begin{enumerate}
  \item Montrer que la limite 
    \begin{equation}
      \lim_{(x,y)\to(0,0)}\frac{x-y^2}{x+y^2},
    \end{equation}
    n'existe pas.
  \item Calculer, si elle existe, la valeur de la limite 
    \begin{equation}
      \lim_{(x,y)\to(0,0)}\frac{x^2-y^2}{x+y}.
    \end{equation}

  \end{enumerate}
