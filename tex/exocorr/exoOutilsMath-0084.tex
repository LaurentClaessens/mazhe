\begin{exercice}\label{exoOutilsMath-0084}

    Soit le champ de vecteurs 
    \begin{equation}
        F(x,y,z)=\frac{1}{ x^2+y^2 }\begin{pmatrix}
            -y    \\ 
            x    \\ 
            4    
        \end{pmatrix}.
    \end{equation}
    \begin{enumerate}
        \item
            Écrire $F$ en coordonnées cylindriques.
        \item
            Donner la divergence de $F$.
    \end{enumerate}
