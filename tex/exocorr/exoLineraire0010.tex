\begin{exercice}\label{exoLineraire0010}

Montrer que
\begin{equation}
	\begin{pmatrix}
	\lambda	&	\lambda	\\ 
	0	&	\lambda	
\end{pmatrix}^n=
\begin{pmatrix}
	\lambda^n	&	n\lambda^n	\\ 
	0	&	\lambda^n	
\end{pmatrix}
\end{equation}
pour tout $\lambda\in\eR$ et tout $n\in\eN_0$.

\corrref{Lineraire0010}
\end{exercice}
