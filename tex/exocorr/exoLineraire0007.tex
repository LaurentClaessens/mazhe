\begin{exercice}\label{exoLineraire0007}

Montrer que les matrices
\begin{equation}
	\begin{aligned}[]
		A&=\begin{pmatrix}
  0	&	1	&	-1\\ 
  4	&	-3	&	4\\ 
 3	&	-3	& 4	  
\end{pmatrix}
	&B&=\begin{pmatrix}
  -1	&	0	&	a\\ 
  0	&	-1	&	b\\ 
 0	&	0	& 1	  
\end{pmatrix}
	\end{aligned},
\end{equation}
avec $a$, $b\in\eR$ sont involutives, c'est-à-dire telles que $X^2=\mtu$.

\end{exercice}
