\begin{exercice}\label{exoFoncDeuxVar0026}

	(INGE1121, 9.2) Démontrer que les fonctions suivantes sont différentiables sur leur domaine de définition.
	\begin{enumerate}

		\item
			\begin{equation}
				f(x,y)=\frac{ xy }{ y^2-x^2 }
			\end{equation}
		\item
			\begin{equation}
				f(x,y,z)=\frac{ xyz }{ x-y }.
			\end{equation}
	\end{enumerate}

\end{exercice}
