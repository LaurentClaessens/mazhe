\begin{exercice}\label{exoINGE1121La0010}

	Pour chacun des systèmes suivants $A\cdot X=B$,
	\begin{enumerate}

		\item
			Résoudre le système par échelonnement,
		\item
			Calculer $A^{-1}$,
		\item
			Vérifier votre réponse en calculant $A^{-1}B$. Qu'êtes-vous censé obtenir ?

	\end{enumerate}

	Les énoncés sont
	\begin{enumerate}

		\item
			\begin{equation}
				\begin{aligned}[]
					A=\begin{pmatrix}
						2	&	1	&	-2	\\
						3	&	2	&	2	\\
						5	&	4	&	3
					\end{pmatrix},
					&&B=\begin{pmatrix}
						10	\\ 
						1	\\ 
						4	
					\end{pmatrix}
				\end{aligned}
			\end{equation}
	\end{enumerate}

\corrref{INGE1121La0010}
\end{exercice}
