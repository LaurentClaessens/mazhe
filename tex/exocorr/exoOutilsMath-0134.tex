\begin{exercice}\label{exoOutilsMath-0134}

    Soit 
    \begin{equation}
        F(x,y)=\begin{pmatrix}
            y+\cos(y)    \\ 
            -x\sin(y)+x    \\ 
            0    
        \end{pmatrix}.
    \end{equation}
    \begin{enumerate}
        \item
            Calculer \( \nabla\times F\)
        \item
            Soit \( \sigma(t)=\big( t,\sin(t^2),1 \big)\) avec \( t\in\mathopen[ 0 , \sqrt{\pi} \mathclose]\). Calculer \( \int_{\sigma}F\).
    \end{enumerate}

\end{exercice}
