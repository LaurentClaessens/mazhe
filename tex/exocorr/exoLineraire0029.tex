\begin{exercice}\label{exoLineraire0029}

	Calculer les déterminants des matrices suivantes
	\begin{equation}
		\begin{aligned}[]
			a&=\begin{pmatrix}
				4	&	5	\\ 
				3	&	7	
			\end{pmatrix}
			&b&=\begin{pmatrix}
				0	&	5	\\ 
				7	&	2	
			\end{pmatrix}\\
			c&=\begin{pmatrix}
				  5	&	3	&	2\\ 
				    1	&	2	&	3\\ 
				     6	&	7	& 9	  
			     \end{pmatrix}
			&d&=\begin{pmatrix}
				  1	&	2	&	3\\ 
				    4	&	5	&	6\\ 
				     7	&	8	& 9	  
			     \end{pmatrix}
		\end{aligned}
	\end{equation}

\end{exercice}
