\begin{exercice}\label{exoOutilsMath-0059}

    Montrer que le champ de vecteurs
    \begin{equation}
        F(x,y)=\begin{pmatrix}
            x^2\sin(xy)    \\ 
            y^2+x\sin(y^2)    \\ 
            0    
        \end{pmatrix}
    \end{equation}
    n'est pas un champ de gradient (c'est à dire qu'il ne dérive pas d'un potentiel).

\end{exercice}
