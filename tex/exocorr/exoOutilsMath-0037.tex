\begin{exercice}\label{exoOutilsMath-0037}

    Montrer qu'il n'existe pas de fonction $f\colon \eR^3\to \eR$ telle que
    \begin{equation}
        \begin{pmatrix}
            \cos(x)    \\ 
            \sin(x)    \\ 
            xy    
        \end{pmatrix}=\nabla f.
    \end{equation}

\end{exercice}
