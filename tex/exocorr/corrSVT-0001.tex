\begin{corrige}{SVT-0001}

    \begin{enumerate}
        \item
            \( \infty\)
        \item
            \( x^2-x^4=x^2(1-x^4)\to -\infty\).
        \item
            \( \infty\)
        \item
            \( 0\)
        \item
            Non déterminé.
        \item
            \( 0\)
        \item
            Par mise en évidence de \( x^2\) sous la racine,
            \begin{equation}
                \frac{ x }{ \sqrt{x^2+1} }=\frac{ x }{ x\sqrt{1+\frac{1}{ x^2 }} }=\frac{1}{ \sqrt{1+\frac{1}{ x^2 }} }\to 1.
            \end{equation}
            La limite est donc \( 1\).
    \end{enumerate}

\end{corrige}
