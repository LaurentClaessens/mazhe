\begin{exercice}\label{exoINGE1121La0016}

	(INGE1121, 1.1) Résoudre les systèmes d'équations suivants:

\begin{multicols}{2}
	\begin{enumerate}

			\item
                                        $
                                        \left\{
                                        \begin{array}{ll}
                                                        x_1 - 2x_2 + 3x_3 - 2x_4 = 0\\ 
                                        3x_1 - 7x_2 - 2x_3 + 4x_4 = 0\\ 
                                        4x_1 + 3x_2 + 5x_3 + 2x_4 = 0\\ 

                                        \end{array}
                                        \right.
                                        $



                                \item
                                        $
                                        \left\{
                                        \begin{array}{ll}
                                                        2x_1 + x_2 - 2x_3 + 3x_4 = 0\\ 
                                        3x_1 + 2x_2 - x_3 + 3x_4 = 4\\ 
                                        3x_1 + 3x_2 + 3x_3 - 3x_4 = 9\\ 

                                        \end{array}
                                        \right.
                                        $



                                \item
                                        $
                                        \left\{
                                        \begin{array}{ll}
                                                        x_1 + 2x_2 - 3x_3 = 0\\ 
                                        2x_1 + 5x_2 + 2x_3 = 0\\ 
                                        3x_1 - x_2 - 4x_3 = 0\\ 

                                        \end{array}
                                        \right.
                                        $



                                \item
                                        $
                                        \left\{
                                        \begin{array}{ll}
                                                        x_1 + 2x_2 - x_3 = 0\\ 
                                        2x_1 + 5x_2 + 2x_3 = 0\\ 
                                        x_1 + 4x_2 + 7x_3 = 0\\ 
                                        x_1 + 3x_2 + 3x_3 = 0\\ 

                                        \end{array}
                                        \right.
                                        $



                                \item
                                        $
                                        \left\{
                                        \begin{array}{ll}
                                                        x_1 + x_2 + x_3 + x_4 = 0\\ 
                                        x_1 + x_2 + x_3 - x_4 = 4\\ 
                                        x_1 + x_2 - x_3 + x_4 = -4\\ 
                                        x_1 - x_2 + x_3 + x_4 = 2\\ 

                                        \end{array}
                                        \right.
                                        $



                                \item
                                        $
                                        \left\{
                                        \begin{array}{ll}
                                                        x_1 + 3x_2 + 3x_3 = 1\\ 
                                        x_1 + 3x_2 + 4x_3 = 0\\ 
                                        x_1 + 4x_2 + 3x_3 = 3\\ 

                                        \end{array}
                                        \right.
                                        $



                                \item
                                        $
                                        \left\{
                                        \begin{array}{ll}
                                                        x_1 - 3x_2 + 2x_3 = -6\\ 
                                        -3x_1 + 3x_2 - x_3 = 17\\ 
                                        2x_1 - x_2 = 3\\ 

                                        \end{array}
                                        \right.
                                        $



                                \item
                                        $
                                        \left\{
                                        \begin{array}{ll}
                                                        x_1 - 2x_2 + 3x_3 - 2x_4 = 0\\ 
                                        3x_1 - 7x_2 - 2x_3 + 4x_4 = 0\\ 
                                        4x_1 + 3x_2 + 5x_3 + 2x_4 = 0\\ 

                                        \end{array}
                                        \right.
                                        $



                                \item
                                        $
                                        \left\{
                                        \begin{array}{ll}
                                                        2x_1 + x_2 - 2x_3 + 3x_4 = 0\\ 
                                        3x_1 + 2x_2 - x_3 + 3x_4 = 4\\ 
                                        3x_1 + 3x_2 + 3x_3 - 3x_4 = 9\\ 

                                        \end{array}
                                        \right.
                                        $



                                \item
                                        $
                                        \left\{
                                        \begin{array}{ll}
                                                        x_1 + 2x_2 - 3x_3 = 0\\ 
                                        2x_1 + 5x_2 + 2x_3 = 0\\ 
                                        3x_1 - x_2 - 4x_3 = 0\\ 

                                        \end{array}
                                        \right.
                                        $



                                \item
                                        $
                                        \left\{
                                        \begin{array}{ll}
                                                        x_1 + 2x_2 - x_3 = 0\\ 
                                        2x_1 + 5x_2 + 2x_3 = 0\\ 
                                        x_1 + 4x_2 + 7x_3 = 0\\ 
                                        x_1 + 3x_2 + 3x_3 = 0\\ 

                                        \end{array}
                                        \right.
                                        $



                                \item
                                        $
                                        \left\{
                                        \begin{array}{ll}
                                                        x_1 + x_2 + x_3 + x_4 = 0\\ 
                                        x_1 + x_2 + x_3 - x_4 = 4\\ 
                                        x_1 + x_2 - x_3 + x_4 = -4\\ 
                                        x_1 - x_2 + x_3 + x_4 = 2\\ 

                                        \end{array}
                                        \right.
                                        $



                                \item
                                        $
                                        \left\{
                                        \begin{array}{ll}
                                                        x_1 + 3x_2 + 3x_3 = 1\\ 
                                        x_1 + 3x_2 + 4x_3 = 0\\ 
                                        x_1 + 4x_2 + 3x_3 = 3\\ 

                                        \end{array}
                                        \right.
                                        $



                                \item
                                        $
                                        \left\{
                                        \begin{array}{ll}
                                                        x_1 - 3x_2 + 2x_3 = -6\\ 
                                        -3x_1 + 3x_2 - x_3 = 17\\ 
                                        2x_1 - x_2 = 3\\ 

                                        \end{array}
                                        \right.
                                        $

	\end{enumerate}


\end{multicols}

\corrref{INGE1121La0016}
\end{exercice}
