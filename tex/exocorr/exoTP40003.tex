\begin{exercice}\label{exoTP40003}

	Calculer la longueur d'arc de la courbe d'équations
	\begin{equation}
		\left\{
		\begin{array}{ll}
			x=e^t\cos(t)\\
			y=e^t\sin(t),
		\end{array}
		\right.
	\end{equation}
	$0\leq t\leq 4\pi$. Dessiner cet arc de courbe.

\end{exercice}
