\begin{exercice}\label{exoINGE1121La0008}

	Soit $A=\begin{pmatrix}
		1	&	1	\\ 
		0	&	1	
	\end{pmatrix}$. Montrer par récurrence que $A^n=\begin{pmatrix}
		1	&	n	\\ 
		0	&	1	
	\end{pmatrix}$.

\corrref{INGE1121La0008}
\end{exercice}
