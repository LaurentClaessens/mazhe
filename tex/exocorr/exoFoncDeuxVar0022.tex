\begin{exercice}\label{exoFoncDeuxVar0022}

	(INGE1121, 9.4) Donner la jacobienne des fonctions suivantes
	\begin{enumerate}

		\item
			\begin{equation}
				f(x,y)=xy\sin(xy)
			\end{equation}
		\item
			\begin{equation}
				f(x,y)=4x-5y
			\end{equation}
		\item
			\begin{equation}
				f(x,y)=\frac{ x }{ 2x+5x }
			\end{equation}
		\item
			\begin{equation}
				f(x,y,z)=\frac{ xy }{ z }
			\end{equation}

	\end{enumerate}

\end{exercice}
