\begin{corrige}{Lineraire0023}

	Dans la base
	\begin{equation}
		\begin{aligned}[]
			\begin{pmatrix}
				7	\\ 
				-5	\\ 
				2	
			\end{pmatrix}&&\text{et}&&\begin{pmatrix}
				\sqrt{3}	\\ 
				29	\\ 
				-8	
			\end{pmatrix},
		\end{aligned}
	\end{equation}
	les deux vecteurs donnés ont évidement les coordonnées $\begin{pmatrix}
		1	\\ 
		0	
	\end{pmatrix}$ et $\begin{pmatrix}
		0	\\ 
		1	
	\end{pmatrix}$.

\end{corrige}
