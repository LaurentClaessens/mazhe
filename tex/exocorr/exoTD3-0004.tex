\begin{exercice}\label{exoTD3-0004}

	Modèle arithmétique. Soit la suite $(u_n)_{n\in\eN}$ définie par
	\begin{equation}
		\begin{cases}
			u_{n+1}=u_n+b	&	\forall n\in\eN_0\\
			u_0=x
		\end{cases}
	\end{equation}
	où $b$ et $x\geq 0$ sont deux nombres réels. Montrer que $u_n=bn+x$ pour tout $n$.

\corrref{TD3-0004}
\end{exercice}
