	\begin{enumerate}
		\item
			Le graphe de $f(x)=2x$ dans $\eR^2$.
		\item
			$\{ (3,y)\tq y\in\eR \}\subset\eR^2$.
		\item
			$\{ (3,y)\tq y\in\mathopen] 1 , 7 \mathclose] \}\subset\eR^2$.
		\item
			$\{ (x,y,0)\tq x\in\eR,\,y\in\mathopen[ 0 , 1 \mathclose] \}\subset\eR^3$.
	\end{enumerate}
