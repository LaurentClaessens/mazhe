\begin{exercice}\label{exoOutilsMath-0131}

    Soit la fonction \( f(x,y)=x^2\sin(y)+ e^{\cos(x)}\).
    \begin{enumerate}
        \item
            Calculer \( df_a(u)\) avec \( a=(\pi,-\pi/4)\) et \( u=(1,5)\).
        \item
            Calculer \( \nabla\cdot(\nabla f)(a)\).
    \end{enumerate}

\end{exercice}
