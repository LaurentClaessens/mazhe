\begin{exercice}\label{exoOutilsMath-0109}

    Soit le champ de vecteurs 
    \begin{equation}
        F(x,y)=\begin{pmatrix}
            y\cos(x)    \\ 
            xy    
        \end{pmatrix}.
    \end{equation}
    Intégrer la fonction $(x,y)\mapsto\nabla\cdot F(x,y,z)$ sur le carré de sommets $(0,0)$, $(1,0)$, $(1,1)$, $(0,1)$.

\corrref{OutilsMath-0109}
\end{exercice}
