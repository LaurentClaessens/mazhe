\begin{exercice}\label{exo_II-1-28}

Résoudre l'équation de Clairaut
\begin{equation}
	y=ty'+\frac{1}{ y'^2 }
\end{equation}
et les problèmes de Cauchy suivants :
\begin{enumerate}
\item $y(1)=2$,
\item $y(2)=3$,
\item $y(0)=1$,
\item $y(0)=-1$,
\item $y(3)=-2$.
\end{enumerate}

\end{exercice}
