\begin{exercice}\label{exo00035}
Si $A+B=\{ a+b\tq a\in A,b\in B \}$, prouver que
\begin{equation}
	\sup(A+B)=\sup(A)+\sup(B).
\end{equation}
\corrref{00035}
\end{exercice}
