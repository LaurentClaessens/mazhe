\begin{exercice}\label{exoEquaDiff0010}

Exercice II.1, page 59. Déterminer la solution générale de chacune des équations différentielles suivantes.
\begin{multicols}{2}
	\begin{enumerate}
		\item
			$y''-2y'-3y=0$
		\item
			$y''+4y'+4y=0$
		\item
			$y''-4y'+5y=0$
		\item
			$y''-4y'+3y=0$
		\item
			$2y''-y'=0$
		\item
			$y''+4y=0$
		\item
			$y''+y'+y=0$
		\item
			$y''+2y'+y=0$
		\item
			$y''-7y'+10y=0$
		\item
			$4y''+12y'+9y=0$
	\end{enumerate}
\end{multicols}
