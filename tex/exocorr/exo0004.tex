\begin{exercice}\label{exo0004}

Soient deux fonctions bornées $f$ et $g$ de $E$ vers $\eR$.
\[
f,g : E \rightarrow \eR
\]
Prouvez que
\[
\sup \{f(x)+g(x) \mid x\in E \} \leq \sup \{f(x) \mid x\in E \} + \sup \{ g(x) \mid x\in E \}
\]
Montrer par un exemple que l'égalité n'a pas toujours lieu.

\corrref{0004}
\end{exercice}
