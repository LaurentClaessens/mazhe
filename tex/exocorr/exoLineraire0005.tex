\begin{exercice}\label{exoLineraire0005}

On donne les matrices
\begin{equation}
	\begin{aligned}[]
		A&=\begin{pmatrix}
  1	&	2	\\ 
  3	&	4	\\ 
 5	&	6	
\end{pmatrix},
	&B&=\begin{pmatrix}
  -3	&	-2	\\ 
  1	&	-5	\\ 
 4	&	3	
\end{pmatrix}.
	\end{aligned}
\end{equation}
Déterminer la matrice $D$ telle que $A+B-D=0$.

\corrref{Lineraire0005}
\end{exercice}
