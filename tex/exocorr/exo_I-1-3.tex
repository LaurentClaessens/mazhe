\begin{exercice}\label{exo113}

Soit 
\begin{equation}
	f_n(x)=
\begin{cases}
	1	&	\text{pour }n<x<n+1\\
	0	&	 \text{sinon.}
\end{cases}
\end{equation}

\begin{enumerate}
\item La suite $\{ f_n \}$ converge-t-elle sur $\eR$ ?
\item La suite $\{ f_n \}$ converge-t-elle uniformément sur $\eR$ ?
\item La suite $\{ f_n \}$ converge-t-elle uniformément sur tout compact de $\eR$ ?
\end{enumerate}

\corrref{113}
\end{exercice}
