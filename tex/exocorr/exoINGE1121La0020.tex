\begin{exercice}\label{exoINGE1121La0020}

	Trouver les valeurs propres et les vecteurs propres de la matrice
	\begin{equation}
		\begin{pmatrix}
			2	&	0	&	\alpha	\\
			0	&	-1	&	0	\\
			0	&	0	&	2
		\end{pmatrix}
	\end{equation}
	où $\alpha$ est un paramètre réel. Discuter selon les valeurs de $\alpha$.

\corrref{INGE1121La0020}
\end{exercice}
