\begin{corrige}{Lineraire0011}

	\begin{enumerate}

		\item
			$(A-B)^2=A^2-AB-BA+b^2$, cela vaut $A^2+B^2$ lorsque $AB=-BA$.
		\item
			$(A+B)(A-B)=A^2-AB+BA-B^2$, cela vaut $A^2-B^2$ lorsque $AB=BA$.

	\end{enumerate}
	

\end{corrige}
