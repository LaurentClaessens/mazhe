\begin{corrige}{OutilsMath-0140}

    L'intégrale à calculer est
    \begin{equation}
        \int_2^3\int_0^{\pi}y\sin(x)dydx=\frac{ \pi^2 }{2}\big(\cos(2)-\cos(3)).
    \end{equation}

\end{corrige}
