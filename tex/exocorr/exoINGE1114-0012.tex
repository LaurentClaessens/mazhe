\begin{exercice}\label{exoSerieUn0012}

	Montrer que si $A$ et $B$ sont des ensembles bornés tels que $A\subseteq B$, alors
	\begin{equation}
		\inf B\leq \inf A\leq\sup A\leq\sup B.
	\end{equation}

\end{exercice}
