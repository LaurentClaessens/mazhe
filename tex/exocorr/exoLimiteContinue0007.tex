\begin{exercice}[\boringexo]\label{exoLimiteContinue0007}

Soit 
\begin{equation}
	\begin{aligned}
		f\colon \eR^2&\to \eR \\
		f(x,y)&=
		\begin{cases}
			\displaystyle\frac{x \ln(1+y)}{e^{xy}-1}	&	\text{si }x\neq  0,y\neq 0\\
            1	&	 \text{si }x=0 \text{ ou } y=0
		\end{cases}
	\end{aligned}
\end{equation}
Calculer $\lim_{(x,y)\to(0,0)}f(x,y)$.

\end{exercice}
