\begin{corrige}{115}
Nous avons que
\begin{equation}
	| f_n(x) |=\frac{ | \sin(nx) | }{ n }\leq \frac{ 1 }{ n }.
\end{equation}
Si $\epsilon>0$ et $N=1/\epsilon$, nous avons que $\forall n>N$,
\begin{equation}
	\frac{ | \sin(nx) | }{ n }\leq \frac{ 1 }{ n }<\frac{1}{ n }<\epsilon,
\end{equation}
ce qui prouve l'uniforme convergence de la suite vers $f=0$.
\end{corrige}
