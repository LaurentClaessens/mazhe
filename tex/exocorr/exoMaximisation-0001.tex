\begin{exercice}\label{exoMaximisation-0001}

	Soit la fonction 
	\begin{equation}
		f(x,y)=\begin{cases}
			\frac{ x^2y^a }{ (x^2+y^2)^2 }	&	\text{si }(x,y)\neq(0,0)\\
			0	&	 \text{sinon}.
		\end{cases}
	\end{equation}
	Prouver que $f$ admet des dérivées partielles seulement si $a>0$.

\corrref{Maximisation-0001}
\end{exercice}
