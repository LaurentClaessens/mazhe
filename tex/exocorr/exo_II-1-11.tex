\begin{exercice}\label{exo_II-1-11}

Résoudre
\begin{enumerate}
\item $y'(t+y^2)+(y-t^2)=0$,
\item $\big(y+ty+\sin(y)\big)dt+\big(t+\cos(y)\big)dy=0$,
\item $y'=-\frac{ 3t^2y+8ty^2 }{ t^3+8t^2y+12y^2 }$.
\end{enumerate}
