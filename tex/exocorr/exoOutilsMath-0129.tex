\begin{exercice}\label{exoOutilsMath-0129}

    \begin{enumerate}
        \item
            Calculer le jacobien de la transformation
            \begin{equation}
                \begin{pmatrix}
                    r    \\ 
                    \theta    \\ 
                    z    
                \end{pmatrix}\mapsto
                \begin{pmatrix}
                    x    \\ 
                    y    \\ 
                    z    
                \end{pmatrix}=\begin{pmatrix}
                    r\cos(\theta)    \\ 
                    r\sin(\theta)    \\ 
                    z    
                \end{pmatrix}.
            \end{equation}
        \item
            En déduire la valeur de l'intégrale \href{http://fr.wikipedia.org/wiki/Fichier:Westmalle_Tripel.jpg}{triple}
            % TODO : il me semble que l'imprimante n'accepte pas les intégrales triples.
            %\begin{equation}
            %    \iiint_D(x^2+y^2+z^2)dxdydz
            %\end{equation}
            \begin{equation}
                \int_D(x^2+y^2+z^2)dxdydz
            \end{equation}
            où \( D\) est le cylindre
            \begin{equation}
                D=\{ (x,y,z); x^2+y^2\leq 1,0\leq z\leq 1 \}.
            \end{equation}

    \end{enumerate}

\end{exercice}
