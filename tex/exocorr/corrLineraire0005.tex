\begin{corrige}{Lineraire0005}

	Si $A+B-D=0$, alors $D=A+B$. Petit calcul :
	\begin{equation}
		D=\begin{pmatrix}
			  1	&	2	\\ 
			    3	&	4	\\ 
			     5	&	6	
		     \end{pmatrix}
		     +
		\begin{pmatrix}
			-3	&	-2	\\
			1	&	-5	\\
			4	&	3	
		\end{pmatrix}
		=
		\begin{pmatrix}
			-2	&	0	\\
			4	&	-1	\\
			9	&	9	
		\end{pmatrix}.
	\end{equation}

\end{corrige}
