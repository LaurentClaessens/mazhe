\begin{exercice}\label{exoSerieUn0011}

	Soit $A$, un ensemble non vide minoré avec $\inf A>0$. Montrer que 
	\begin{equation}
		\sup\{ \frac{1}{ x }\tq x\in A \}=\frac{1}{ \inf A }.
	\end{equation}

\end{exercice}
