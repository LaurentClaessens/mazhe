\begin{exercice}\label{exo0083}

Montrer que pour tout $x \in \eR^n$ et tout $r >0$ la boule 
\[ 
	B(x,r):=\big\{ y \in \eR^n \mid \left|x-y \right| < r \big\}
\]
est ouverte.

\corrref{0083}
\end{exercice}
