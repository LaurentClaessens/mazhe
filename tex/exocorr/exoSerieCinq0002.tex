\begin{exercice}\label{exoSerieCinq0002}

Soit $A$ une matrice carrée de taille $N$. Démontrer que 
\begin{equation}
	\det A=\sum_{\sigma\in\Sym(N)} \epsilon(\sigma)A_{1\sigma(1)}...A_{N\sigma(N)}.
\end{equation}

\end{exercice}
