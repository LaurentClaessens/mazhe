\begin{exercice}\label{exoVariete0005}

Trouver les extrema de la fonction 
\begin{equation}
	\begin{aligned}
		f\colon \eR^3&\to \eR \\
		(x,y,z)&\mapsto x^2+y^2+z^2 
	\end{aligned}
\end{equation}
relativement à la surface $S := \{ z^2 = x^2 + y^2 + xy + x + y + z + 1 \}$.


\corrref{Variete0005}
\end{exercice}
