\begin{corrige}{0039}

Posons $g(x) = \abs x$ et
\begin{equation*}
  f(x) =
  \begin{cases}
    -1 & \text{si }x \in \eQ\\
    1 &  \text{sinon.}\\
  \end{cases}
\end{equation*}
alors $g(f(x)) = 1$ pour tout $x$, donc $g\circ f$ est continue.

\end{corrige}
