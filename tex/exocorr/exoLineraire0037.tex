\begin{exercice}\label{exoLineraire0037}

	Exercice 14, page 91. Soit 
	\begin{equation}
		\begin{aligned}[]
			A_1&=\begin{pmatrix}
				3	&	-1	&	1\\ 
				-1	&	5	&	-1\\ 
				1	&	-1	&	3	  
			\end{pmatrix}\\
			A_2&=\begin{pmatrix}
				2	&	0	&	1\\ 
				0	&	3	&	0\\ 
				1	&	0	&	2	  
			\end{pmatrix}.
		\end{aligned}
	\end{equation}
	Trouver une matrice orthogonale $P$ qui transforme $A$ en une matrice diagonale.

\corrref{Lineraire0037}
\end{exercice}
