\begin{exercice}\label{exoINGE1121La0019}

	(INGE1121, 6.7) Déterminer le genre des formes quadratiques associées aux matrices suivantes en utilisant au maximum les propositions relatives aux mineurs, mineurs principaux primaires, \ldots

	\begin{multicols}{2}
		\begin{enumerate}
			\item
				\begin{equation}
					A=\begin{pmatrix}
						-6	&	1	&	-2	\\
						1	&	-3	&	1	\\
						-2	&	1	&	-1
					\end{pmatrix}
				\end{equation}
			\item
				\begin{equation}
					A=\begin{pmatrix}
						5	&	2	&	10	\\
						2	&	-7	&	3	\\
						10	&	3	&	0
					\end{pmatrix}.
				\end{equation}
				
		\end{enumerate}
	\end{multicols}
	

\corrref{INGE1121La0019}
\end{exercice}
