\begin{corrige}{OutilsMath-0014}

	Nommons $S$ le cercle de centre $(-1,2)$ et de rayon $1$. Nous avons $(x,y)\in S$ si et seulement si la distance entre $(x,y)$ et $(-1,2)$ vaut exactement $1$ :
	\begin{equation}
		f\big( (x,y),(-1,2) \big)=1.
	\end{equation}
	L'équation du cercle $S$ est donc
	\begin{equation}
		(-1-x)^2+(2-y)^2=1^2,
	\end{equation}
	c'est-à-dire
	\begin{equation}
		(x+1)^2+(y-2)^2=1.
	\end{equation}

\end{corrige}
