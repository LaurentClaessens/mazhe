\begin{exercice}\label{exo0082}

Soit $(X,d)$ un espace métrique. Prouver l'inégalité quadrilatère
\begin{equation}
	\left| d(a,b)-d(u,v) \right| \,\le\, d(a,u) + d(b,v)
\end{equation}
pour tout $a,b,u,v \in X$.

\corrref{0082}
\end{exercice}
