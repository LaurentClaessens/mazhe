\begin{exercice}\label{exoVariete0006}

Calculez la longueur des arcs de courbe suivants:
\begin{enumerate}
	\item
		$y= \ln(1-x^2)$ pour  $0\leq x\leq \frac{1}{2}$
	\item
		$y= x^{3/2}$ pour  $0\leq x\leq 5$
	\item
		$y = 1-\ln(\cos x)$ pour $0\leq x \leq \frac{\pi}{4}$
\end{enumerate}

\end{exercice}
