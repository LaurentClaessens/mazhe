	      \begin{equation*}
		      \int_{\eR} e^{-x^2}dx=\sqrt{\pi }

	\item
	      Le lemme~\ref{LEMooTDWSooSBJXdv} donne une suite régularisante à base de gaussienne.

\end{enumerate}
