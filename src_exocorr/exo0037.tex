\begin{exercice}\label{exo0037}

Montrez que la fonction
\begin{equation}
	\begin{aligned}
		f\colon \eR&\to \eR \\
		x&\mapsto f(x)=\begin{cases}
	1	&	\text{si $x\in\eQ$}\\
	0	&	 \text{si $x\neq\eQ$}
\end{cases}
\end{aligned}
\end{equation}
est discontinue sur $\mathbb{R}$ et que la fonction $f\colon \eQ\to \eR$, $x\mapsto 1$ est continue sur $\eQ$.

\end{exercice}
