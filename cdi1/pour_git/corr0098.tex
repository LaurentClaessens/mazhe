\begin{corrige}{0098}

La fonction 
\begin{equation}
	f(x)=\begin{cases}
	x	&	\text{si $x<1$}\\
	3-x	&	 \text{si $1<x\leq2$}
\end{cases}
\end{equation}
Elle vérifie $f^{-1}(1)=2$, mais $\lim_{y\to 1^-}f^{-1}(y)=1$, ce qui prouve que $f^{-1}$ n'est pas continue.

\end{corrige}
